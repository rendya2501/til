            var query = new StringBuilder()
                .AppendLine("SELECT")
                .AppendLine("    0 AS [IsSelectedCancel],")
                .AppendLine("    CASE WHEN [TFr_Slip].[SettlementID] IS NULL THEN 0 ELSE 1 END AS [IsSettled],")
                .AppendLine("    [TCs_AccountsReceivableDetail].[SettlementID],")
                .AppendLine("    [TRe_ReservationPlayer].[AccountNo],")
                .AppendLine("    [TRe_ReservationPlayer].[Name] AS [AttendeeName],")
                .AppendLine("    [TCs_AccountsReceivableDetail].[AccountName],")
                .AppendLine("    [TCs_AccountsReceivableDetail].[DepositAmount] AS [AccountsReceivable]")
                .AppendLine("FROM")
                .AppendLine("    [TFr_Slip]")
                .AppendLine("    JOIN [TRe_ReservationPlayer]")
                .AppendLine("    ON [TFr_Slip].[OfficeCD] = [TRe_ReservationPlayer].[OfficeCD]")
                .AppendLine("    AND [TFr_Slip].[SettlementPlayerNo] = [TRe_ReservationPlayer].[PlayerNo]")
                .AppendLine("    JOIN [TCs_AccountsReceivablePayment]")
                .AppendLine("    ON [TFr_Slip].[OfficeCD] = [TCs_AccountsReceivablePayment].[OfficeCD]")
                .AppendLine("    AND CONCAT([TFr_Slip].[SlipID],FORMAT([TFr_Slip].[DetailNo],'000000')) = [TCs_AccountsReceivablePayment].[PaymentSlipID]")
                .AppendLine("    JOIN [TCs_AccountsReceivablePaymentSettleIDList]")
                .AppendLine("    ON [TCs_AccountsReceivablePayment].[AccountsReceivablePaymentID] = [TCs_AccountsReceivablePaymentSettleIDList].[AccountsReceivablePaymentID]")
                .AppendLine("    JOIN [TCs_AccountsReceivableDetail]")
                .AppendLine("    ON [TCs_AccountsReceivablePaymentSettleIDList].[SettlementID] = [TCs_AccountsReceivableDetail].[SettlementID]")
                .AppendLine("WHERE")
                .AppendLine("    [TFr_Slip].[SlipID] = @slipID");
            return _DapperAction.GetDataListByQuery<AccountsReceivablePaymentListItem>(
                ConnectionTypes.Data,
                query.ToString(),
                new { slipID = condition.SlipID }
