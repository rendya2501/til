``` sql
SELECT DISTINCT
    ReservationFrameNo,
    FIRST_VALUE([SQ].[Numerator] + '/' + [SQ].[Denominator]) OVER(PARTITION BY ReservationFrameNo ORDER BY ReservationNo)
        + CASE WHEN COUNT(*) OVER (PARTITION BY [ReservationFrameNo]) > 1 THEN '+' ELSE '' END AS [Result]
    --CASE WHEN COUNT(*) > 1 THEN '+' ELSE '' END  AS [FrameCount],
FROM (
    SELECT
        [SQ_Numerator].ReservationFrameNo,
        [SQ_Numerator].ReservationNo,
        CONVERT(nvarchar,[SQ_Numerator].[Numerator]) AS [Numerator],
        CONVERT(nvarchar,[SQ_Denominator].[Denominator]) AS [Denominator]
    FROM 
        -- 予約枠を上から見ていった時のシーケンス(分子:Numerator)
        (
            SELECT
                [ReservationFrameNo],
                [ReservationNo],
                ROW_NUMBER() OVER(PARTITION BY [ReservationNo] ORDER BY ReservationFrameNo) AS Numerator
            FROM 
                ReservationPlayer
            GROUP BY
                ReservationFrameNo,ReservationNo
        ) AS [SQ_Numerator]
    JOIN 
        -- 予約が持っている予約枠数(分母:Denominator)
        (
            SELECT
                [ReservationNo],
                COUNT(DISTINCT [ReservationFrameNo]) AS Denominator
            FROM 
                ReservationPlayer
            GROUP BY 
                [ReservationNo]
        ) AS [SQ_Denominator]
    ON [SQ_Numerator].ReservationNo = [SQ_Denominator].ReservationNo
) AS [SQ]



SELECT

        ROW_NUMBER() OVER(PARTITION BY ReservationFrameNo ORDER BY ReservationNo) rk
    FROM (
        SELECT
            [SQ_Numerator].ReservationFrameNo,
            [SQ_Numerator].ReservationNo,

        FROM 
            -- 予約枠を上から見ていった時のシーケンス(分子:Numerator)
            (
                SELECT
                    [ReservationFrameNo],
                    [ReservationNo],
                    ROW_NUMBER() OVER(PARTITION BY [ReservationNo] ORDER BY ReservationFrameNo) AS Numerator
                FROM 
                    ReservationPlayer
                GROUP BY
                    ReservationFrameNo,ReservationNo
            ) AS [SQ_Numerator]
        JOIN 
            -- 予約が持っている予約枠数(分母:Denominator)
            (
                SELECT
                    [ReservationNo],
                    COUNT(DISTINCT [ReservationFrameNo]) AS Denominator
                FROM

SELECT
    [SQ].[ReservationFrameNo],
    [SQ].[Numerator] + '/' + [SQ].[Denominator] + CASE
        WHEN [SQ].[Count] > 1 THEN '+'
        ELSE ''
    END AS [FrameCount]
FROM
    (
        SELECT
            [SQ_Numerator].[ReservationFrameNo],
            [SQ_Numerator].[ReservationNo],
            [SQ_SeatNo].[SeatNo],
            COUNT(*) OVER(PARTITION BY [SQ_Numerator].[ReservationFrameNo]) AS [Count],
            CONVERT(nvarchar, [SQ_Numerator].[Numerator]) AS [Numerator],
            CONVERT(nvarchar, [SQ_Denominator].[Denominator]) AS [Denominator]
        FROM
            (
                SELECT
                    [ReservationFrameNo],
                    [ReservationNo],
                    ROW_NUMBER() OVER(PARTITION BY [ReservationNo] ORDER BY [ReservationFrameNo]) AS [Numerator]
                FROM
                    ReservationPlayer
                GROUP BY
                    [ReservationFrameNo],
                    [ReservationNo]
            ) AS [SQ_Numerator]
            INNER JOIN
                (
                    SELECT
                        [ReservationNo],
                        COUNT(DISTINCT [ReservationFrameNo]) AS [Denominator]
                    FROM
                        ReservationPlayer
                    GROUP BY
                        [ReservationNo]
                ) AS [SQ_Denominator]
            ON  [SQ_Numerator].[ReservationNo] = [SQ_Denominator].[ReservationNo]
            INNER JOIN
                (
                    SELECT
                        [ReservationFrameNo],
                        [ReservationNo],
                        MIN([SeatNo]) AS [SeatNo]
                    FROM
                        ReservationPlayer
                    GROUP BY
                        [ReservationFrameNo],
                        [ReservationNo]
                ) AS [SQ_SeatNo]
            ON  [SQ_Numerator].[ReservationNo] = [SQ_SeatNo].[ReservationNo]
            AND [SQ_Numerator].[ReservationFrameNo] = [SQ_SeatNo].[ReservationFrameNo]
    ) AS [SQ]
WHERE
    [SQ].[SeatNo] = 1




SELECT DISTINCT
    [SQ].ReservationFrameNo,
     FIRST_VALUE([SQ].[Result]) OVER (PARTITION BY ReservationFrameNo ORDER BY ReservationNo) AS [Result]
FROM(
    SELECT   
        ReservationFrameNo,
        [ReservationNo],
        CONVERT(nvarchar,ROW_NUMBER() OVER(PARTITION BY [ReservationNo] ORDER BY [ReservationFrameNo]))
        + '/' 
        + CONVERT(nvarchar,COUNT(1) OVER(PARTITION BY [ReservationNo]))
        + CASE WHEN COUNT(*) OVER (PARTITION BY [ReservationFrameNo]) > 1 THEN '+' ELSE '' END  AS [Result]
    FROM ReservationPlayer
    GROUP BY [ReservationFrameNo],[ReservationNo]
) AS [SQ]
```

## 分子分母を出力するサンプル

``` sql
SELECT
    ReservationFrameNo,
    [Numerator] + '/' + [Denominator] + CASE WHEN COUNT(*) > 1 THEN '+' ELSE '' END  AS [FrameCount]
FROM (
    SELECT
        [SQ_Numerator].ReservationFrameNo,
        [SQ_Numerator].ReservationNo,
        CONVERT(nvarchar,[SQ_Numerator].[Numerator]) AS [Numerator],
        CONVERT(nvarchar,[SQ_Denominator].[Denominator]) AS [Denominator]
    FROM (
    -- 予約枠を上から見ていった時のシーケンス(分子:Numerator)
        SELECT
            [ReservationFrameNo],
            [ReservationNo],
            ROW_NUMBER() OVER(PARTITION BY [ReservationNo] ORDER BY ReservationFrameNo) AS Numerator
        FROM 
            [ReservationPlayer]
        GROUP BY
            ReservationFrameNo,ReservationNo
        ) AS [SQ_Numerator]
    JOIN (
    -- 予約が持っている予約枠数(分母:Denominator)
        SELECT
            [ReservationNo],
            COUNT(DISTINCT [ReservationFrameNo]) AS Denominator
        FROM 
            [ReservationPlayer]
        GROUP BY 
            [ReservationNo]
    ) AS [SQ_Denominator]
    ON [SQ_Numerator].ReservationNo = [SQ_Denominator].ReservationNo
) AS [SQ]
GROUP BY
    [ReservationFrameNo]
```

## グループ毎に1行だけ取り出したけど、それぞれ何行がひとまとまりになったのか表示する

`COUNT(フィールド名) OVER ( PARTITION BY フィールド名 )`  

RANK,ROW_NUMBER,COUNTにOVERを使えるあたり、OVERはOVERでそういう構文になっているのだろうか。  

組数を表示するときに使った。  
予約枠でサマった時に、その枠に2つ以上の予約があれば「分子/分母 ＋」と表示させたい。  
でも、わざわざGROUP BY COUNTしたサブクエリを作りたくない。  
その時に活躍した。  

実務において使用したのでメモ。  
例えば、その時間に複数の予約が入っている場合、
その時間に2人以上がいる場合は、「+」と表示させたい。
